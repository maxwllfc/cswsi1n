            # Função para remover outliers usando o desvio padrão
            def remove_outliers(data, threshold=3):
                # Calcula a média e o desvio padrão dos dados
                mean = np.mean(data)
                std_dev = np.std(data)
                
                # Identifica os outliers (valores fora do intervalo de [mean - threshold*std, mean + threshold*std])
                filtered_data = [x for x in data if (mean - threshold * std_dev) <= x <= (mean + threshold * std_dev)]
                
                return filtered_data
            
            # Exemplo de dados com outliers
            dados = [10, 12, 15, 16, 100, 17, 18, 14, 13, 11, 120]
            
            # Remover outliers
            dados_sem_outliers = remove_outliers(dados)
            
            # Exibe os resultados
            print("Dados originais:", dados)
            print("Dados sem outliers:", dados_sem_outliers)
